## Solution

```java
import java.io.*;
import java.util.*;

public class Solution {
    public static void main(String[] args) {
        int count = 1;
        String str = "";
        Scanner scanner = new Scanner(System.in);

        while (scanner.hasNext()) {
            str = scanner.nextLine();
            System.out.println(count + " " + str);
            count++;
        }

        scanner.close();
    }
```
