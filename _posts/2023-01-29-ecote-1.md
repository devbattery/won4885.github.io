## 상하좌우
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Solution {
    public static void solution(int n, String[] plans) {
        // 기본 좌표 (1, 1) 설정
        int x = 1, y = 1;

        // moveTypes에 따른 이동 방향 설정
        int[] dx = {0, 0, -1, 1};
        int[] dy = {-1, 1, 0, 0};
        String[] moveTypes = {"L", "R", "U", "D"};

        // 이동 계획을 하나씩 확인
        for (String plan : plans) {
            // 이동 후 좌표 구하기
            int nx = -1, ny = -1;
            for (int i = 0; i < moveTypes.length; i++) {
                if (plan.equals(moveTypes[i])) {
                    nx = x + dx[i];
                    ny = y + dy[i];
                }
            }

            // 공간을 벗어나는 경우 무시
            if (nx < 1 || ny < 1 || nx > n || ny > n) {
                continue;
            }

            // 이동을 수행
            x = nx;
            y = ny;
        }

        System.out.print(x + " " + y);
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int n = Integer.parseInt(br.readLine());
        String[] plans = br.readLine().split(" ");
        solution(n, plans);
        br.close();
    }
}
```

## 시각
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Solution {
    private static final int SIXTY = 60;

    public static int solution(int n) {
        int cnt = 0;

        for (int h = 0; h < n + 1; h++) {
            for (int m = 0; m < SIXTY; m++) {
                for (int s = 0; s < SIXTY; s++) {
                    // 특정한 시각에 3이 포함되어 있는가?
                    if (h % 10 == 3
                            || m / 10 == 3 || m % 10 == 3
                            || s / 10 == 3 || s % 10 == 3) {
                        cnt++;
                    }
                }
            }
        }

        return cnt;
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        System.out.print(solution(Integer.parseInt(br.readLine())));
        br.close();
    }
}
```
